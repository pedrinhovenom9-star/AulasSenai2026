      function TestaAptoDoar() {
        var nome = document.getElementById("nome").value;
        var peso = Number(document.getElementById("peso").value);
        var idade = Number(document.getElementById("idade").value);
        var genero = document.getElementById("genero").value;

        var resultado = document.getElementById("resultado");

        if (peso >= 50 && idade >= 18 && idade <= 69) {
          if (genero === "Masculino") {
            resultado.innerHTML = `Olá ${nome}, você deve doar 700 gramas 🩸`;
          } else if (genero === "Feminino") {
            resultado.innerHTML = `Olá ${nome}, você deve doar 400 gramas 🩸`;
          } else {
            resultado.innerHTML = `Selecione um gênero 🙂`;
          }
        } else {
          resultado.innerHTML = `Lamento ${nome}, você não está apto para a doação 😢<br>Obrigado pela boa intenção!`;
        }
      }
